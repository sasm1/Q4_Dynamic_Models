Taking into account the hint, we should estimate the parameters using the EM algortihm with four different assumptions of the DGP:
  • Two Normal distributions
  • Two Extreme Value distributions
  • Three Normal distributions
  • Three Extreme Value distributions

EM Algorithm Two Normal distributions

% Step 1 (a): initial values for each distribution and the mixture
obs = size(Y,1);
theta_0_1 = [4,2.5]; % [mu1, sigma1]
theta_0_2 = [6,2.5]; % [mu2, sigma2]
pi_0 = [0.5,0.5]; % Initial mixture weights
 
% Step 1 (b): specify convergence criteria
iter = 1;
maxiter = 1000;
lik0 = -Inf;
normdiff = Inf;
tolerance = 1e-6;
 
while normdiff > tolerance && iter <= maxiter
% Expectation step
pdf1 = normpdf(Y, theta_0_1(1), theta_0_1(2));
pdf2 = normpdf(Y, theta_0_2(1), theta_0_2(2));
EZ = zeros(obs, 2);
EZ(:,1) = (pi_0(1) * pdf1) ./ (pi_0(1)*pdf1 + pi_0(2)*pdf2);
EZ(:,2) = 1 - EZ(:,1); % Complement for expected Z
% Maximization step
theta_fun1 = @(theta) -sum(log(normpdf(Y, theta(1), abs(theta(2)))) .* EZ(:,1));
theta_fun2 = @(theta) -sum(log(normpdf(Y, theta(1), abs(theta(2)))) .* EZ(:,2));
[theta_1, ~] = fminsearch(theta_fun1, theta_0_1);
[theta_2, ~] = fminsearch(theta_fun2, theta_0_2);
% Check for positive standard deviations
theta_1(2) = abs(theta_1(2));
theta_2(2) = abs(theta_2(2));
pi_1 = mean(EZ);
 
% Update likelihood:
pdf1_new = normpdf(Y, theta_1(1), theta_1(2));
pdf2_new = normpdf(Y, theta_2(1), theta_2(2));
lik1 = sum(log(pi_1(1)*pdf1_new + pi_1(2)*pdf2_new));
normdiff = (lik1 - lik0)^2;
% Update for next iteration
lik0 = lik1;
theta_0_1 = theta_1;
theta_0_2 = theta_2;
pi_0 = pi_1;
iter = iter + 1;
end
 
ThetaHat1 = theta_0_1;
ThetaHat2 = theta_0_2;
PiHat = pi_0;
